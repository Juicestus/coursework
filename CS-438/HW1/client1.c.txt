#include <stdio.h>
#include <stdlib.h>
#include <rpc/rpc.h>
#include "strings.h"

int main(int argc, char** argv) {
    CLIENT *clnt;
    enum clnt_stat stat;
    char *server;
    char *input_str;
    char **result;
    stuff out;

    if (argc < 3) {
        fprintf(stderr, "usage: %s <host> <string>\n", argv[0]);
        exit(1);
    }

    server = argv[1];    // e.g., "localhost"
    input_str = argv[2]; // The string to send

    /* 1. Create a client handle */
    /* "udp" is chosen here since your original code used broadcast (which is UDP only) */
    clnt = clnt_create(server, TEST_PROG, TEST_VERS, "udp");

    if (clnt == NULL) {
        clnt_pcreateerror(server);
        exit(1);
    }

    /* 2. Prepare the data */
    out.name = input_str;
    out.val = 1234;

    /* 3. Perform the Remote Procedure Call */
    /* We use the specific function name generated by rpcgen (usually test_func_1) */
    result = test_func_1(&out, clnt);

    if (result == (char **) NULL) {
        clnt_perror(clnt, "call failed");
        exit(1);
    }

    /* 4. Use the result */
    printf("Server responded: %s\n", *result);

    /* 5. Cleanup */
    clnt_destroy(clnt);
    return 0;
}
