#include <stdio.h>
#include <threading.h>
#include <assert.h> // For assert()

/**
 * Initializes the threading library.
 * This function must be called before any other t_* function.
 * It sets up the context for the main thread and initializes all other
 * context slots to be invalid.
 */
void t_init()
{
    // Initialize all context slots to INVALID
    for (int i = 0; i < NUM_CTX; i++)
    {
        contexts[i].state = INVALID;
    }

    // The 'main' function is our first thread, conventionally at index 0
    contexts[0].state = VALID;
    current_context_idx = 0;
}

/**
 * Creates a new thread to execute the given function.
 *
 * @param foo The function pointer for the new thread to execute.
 * @param arg1 The first integer argument to the function.
 * @param arg2 The second integer argument to the function.
 * @return 0 on success, 1 on failure (e.g., no free context slots).
 */
int32_t t_create(fptr foo, int32_t arg1, int32_t arg2)
{
    int new_idx = -1;
    // Find an empty slot for the new context, starting from index 1
    for (int i = 1; i < NUM_CTX; i++)
    {
        if (contexts[i].state == INVALID)
        {
            new_idx = i;
            break;
        }
    }

    // If no free slot is found, return an error
    if (new_idx == -1)
    {
        return 1;
    }

    // Get the current context as a template for the new one
    getcontext(&contexts[new_idx].context);

    // Allocate a new stack for the thread
    char *stack = (char *)malloc(STK_SZ);
    if (stack == NULL)
    {
        return 1; // Malloc failed
    }

    // Configure the new context's stack
    contexts[new_idx].context.uc_stack.ss_sp = stack;
    contexts[new_idx].context.uc_stack.ss_size = STK_SZ;
    contexts[new_idx].context.uc_stack.ss_flags = 0;
    
    // Set the successor context to NULL. If foo() returns without calling
    // t_finish(), the process will exit. Our design requires t_finish().
    contexts[new_idx].context.uc_link = NULL;

    // Modify the context to execute the 'foo' function with its arguments
    makecontext(&contexts[new_idx].context, (void (*)())foo, 2, arg1, arg2);
    
    // Mark the context as valid and ready to be scheduled
    contexts[new_idx].state = VALID;

    return 0; // Success
}

/**
 * Yields the CPU to the next available thread in a round-robin fashion.
 *
 * @return The number of other VALID threads remaining, or 0 if none.
 */
int32_t t_yield()
{
    // Find the next valid context to switch to
    int next_idx = -1;
    for (int i = 1; i < NUM_CTX; i++)
    {
        int idx = (current_context_idx + i) % NUM_CTX;
        if (contexts[idx].state == VALID)
        {
            next_idx = idx;
            break;
        }
    }

    // If no other valid context is found, there's nothing to yield to.
    if (next_idx == -1 || next_idx == current_context_idx)
    {
        return 0;
    }

    // Store the current context index before switching
    uint8_t old_idx = current_context_idx;
    current_context_idx = next_idx;
    
    // Save the current context and switch to the next one
    swapcontext(&contexts[old_idx].context, &contexts[next_idx].context);
    
    // --- Execution resumes here when another thread yields back to this one ---
    
    // After resuming, calculate the number of other valid threads
    int valid_count = 0;
    for (int i = 0; i < NUM_CTX; i++)
    {
        if (i == current_context_idx) continue;
        if (contexts[i].state == VALID)
        {
            valid_count++;
        }
    }
    
    return valid_count;
}

/**
 * Called by a thread to terminate itself.
 * It frees the thread's resources and schedules the next available thread.
 * This function does not return.
 */
void t_finish()
{
    uint8_t finished_idx = current_context_idx;

    // Free the stack allocated for this thread
    assert(contexts[finished_idx].context.uc_stack.ss_sp != NULL);
    free(contexts[finished_idx].context.uc_stack.ss_sp);

    // Mark the context as finished and invalid
    contexts[finished_idx].state = DONE;

    // Find the next valid thread to run
    int next_idx = -1;
    for (int i = 1; i < NUM_CTX; i++)
    {
        int idx = (finished_idx + i) % NUM_CTX;
        if (contexts[idx].state == VALID)
        {
            next_idx = idx;
            break;
        }
    }

    // There must be another thread (at least main) to switch to.
    // If not, the program has a logic error (e.g., main terminated early).
    assert(next_idx != -1);

    // Update the global index and switch to the next context permanently.
    // setcontext() does not save the current context and never returns.
    current_context_idx = next_idx;
    setcontext(&contexts[next_idx].context);
    
    // This part should be unreachable
    perror("setcontext failed");
    exit(1);
}
